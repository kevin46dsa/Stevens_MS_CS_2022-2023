?rpart()
CART_class <- rpart(Class~., data = train_data)

# ?predict()
prediction <- predict(CART_class, test_data, type = "class")

#generating matrix
conf_matrix <- table(prediction,test_data$Class)

accuracy <- function(x){sum(diag(x)/(sum(rowSums(x)))) * 100}
